""" OPEN Pre-Run Pattern Scanner — FULL DYNAMIC VERSION (Robust Pre-Filter) Fingerprints OPEN's pre-run and scans current Russell 2000 small-caps dynamically via IWM ETF holdings CSV. """ import yfinance as yf import pandas as pd import numpy as np import requests from io import StringIO from datetime import datetime, timedelta import warnings import time warnings.filterwarnings('ignore') # ───────────────────────────────────────────── # CONFIG # ───────────────────────────────────────────── OPEN_PRE_RUN_START = "2024-06-01" OPEN_PRE_RUN_END = "2024-10-01" LOOKBACK_DAYS = 400 TODAY = datetime.today().date() ONE_YEAR_AGO = TODAY - timedelta(days=LOOKBACK_DAYS) PRICE_MAX = 25 PRICE_MIN = 1 VOL_MIN_PCT = 60 CORR_MIN = 0.25 MAX_TICKERS_TO_SCAN = 300 # increase to 500+ if you want broader scan (slower) IWM_HOLDINGS_URL = "https://www.ishares.com/us/products/239710/ishares-russell-2000-etf/1467271812596.ajax?fileType=csv&fileName=IWM_holdings&dataType=fund" # ───────────────────────────────────────────── # DYNAMIC UNIVERSE: Fetch & Parse Russell 2000 from IWM CSV # ───────────────────────────────────────────── def get_russell_2000_tickers(): print("Fetching current Russell 2000 holdings from iShares IWM ETF CSV...") try: response = requests.get(IWM_HOLDINGS_URL, timeout=20) response.raise_for_status() csv_text = response.text lines = csv_text.splitlines() skip_rows = 0 header_found = False for i, line in enumerate(lines): stripped = line.strip() if stripped.startswith('"Ticker"') or 'Ticker' in stripped or 'Symbol' in stripped: skip_rows = i header_found = True break if not header_found: raise ValueError("Could not locate header row containing 'Ticker' or 'Symbol'") csv_data = StringIO('\n'.join(lines[skip_rows:])) df = pd.read_csv(csv_data, on_bad_lines='skip') df.columns = df.columns.str.strip().str.replace('"', '').str.replace("'", "") ticker_col = None possible_cols = ['Ticker', 'Symbol', 'Issuer Ticker', 'Holding Ticker'] for col in possible_cols: if col in df.columns: ticker_col = col break if ticker_col is None: raise ValueError(f"No ticker column found. Columns: {list(df.columns)}") tickers = df[ticker_col].dropna().astype(str).str.strip().tolist() tickers = [t for t in tickers if t.isalnum() and 1 <= len(t) <= 5 and not t.isdigit()] print(f"Successfully parsed {len(tickers)} tickers from IWM holdings CSV.") return list(set(tickers)) except Exception as e: print(f"Failed to fetch/parse IWM CSV: {type(e).__name__}: {str(e)}") print("URL:", IWM_HOLDINGS_URL) return [] # ───────────────────────────────────────────── # Pre-filter with robust error handling # ───────────────────────────────────────────── def pre_filter_tickers(tickers): print(f"Pre-filtering {len(tickers)} tickers (price ${PRICE_MIN}-${PRICE_MAX}, small/mid-cap)...") filtered = [] skipped = 0 for ticker in tickers: try: info = yf.Ticker(ticker).info or {} # Try multiple possible price keys price_keys = ['regularMarketPrice', 'previousClose', 'currentPrice', 'regularMarketPreviousClose', 'fiftyTwoWeekHigh'] price = None for key in price_keys: if info.get(key) is not None: price = float(info[key]) break mcap = info.get('marketCap', 0) if price is not None and PRICE_MIN <= price <= PRICE_MAX and 300000000 <= mcap <= 10000000000: filtered.append(ticker) time.sleep(0.4) # prevent rate limiting if len(filtered) >= MAX_TICKERS_TO_SCAN * 1.2: break except Exception: skipped += 1 # Uncomment next line for debug: # print(f"Skipped {ticker} (data error)") continue print(f"Pre-filtered to {len(filtered)} candidates ({skipped} skipped due to data errors).") return filtered[:MAX_TICKERS_TO_SCAN] # ───────────────────────────────────────────── # INDICATOR FUNCTIONS # ───────────────────────────────────────────── def compute_rsi(prices, period=14): delta = prices.diff() gain = delta.clip(lower=0) loss = -delta.clip(upper=0) avg_gain = gain.ewm(com=period - 1, min_periods=period).mean() avg_loss = loss.ewm(com=period - 1, min_periods=period).mean() rs = avg_gain / avg_loss.replace(0, np.nan) rsi = 100 - (100 / (1 + rs)) return rsi def compute_atr(hist, period=14): high, low, close = hist['High'], hist['Low'], hist['Close'] prev_close = close.shift(1) tr = pd.concat([high - low, (high - prev_close).abs(), (low - prev_close).abs()], axis=1).max(axis=1) return tr.rolling(period).mean() def volume_surge_score(hist, lookback=20): vol = hist['Volume'] if len(vol) < lookback + 5: return 0.0 avg_recent = vol.iloc[-5:].mean() avg_base = vol.iloc[-lookback:-5].mean() if avg_base <= 0 or np.isnan(avg_base): return 0.0 return avg_recent / avg_base def rsi_buildup_score(rsi_series): if rsi_series.dropna().empty or len(rsi_series.dropna()) < 3: return 0.0 recent_rsi = rsi_series.dropna().iloc[-5:] current_rsi = recent_rsi.iloc[-1] rising = recent_rsi.iloc[-1] > recent_rsi.iloc[-3] in_zone = 45 <= current_rsi <= 72 score = 0.0 if in_zone: score += 0.5 if rising: score += 0.3 if current_rsi > 55: score += 0.2 return min(score, 1.0) def compression_score(hist, atr_series, lookback=20): if atr_series.dropna().empty or len(atr_series.dropna()) < lookback: return 0.0 recent_atr = atr_series.dropna().iloc[-5:].mean() baseline_atr = atr_series.dropna().iloc[-lookback:-5].mean() if baseline_atr <= 0 or np.isnan(baseline_atr): return 0.0 compression_ratio = recent_atr / baseline_atr score = max(0, 1 - compression_ratio) return min(score, 1.0) # ───────────────────────────────────────────── # FINGERPRINT OPEN # ───────────────────────────────────────────── def get_open_fingerprint(): print("=" * 60) print("FINGERPRINTING OPEN'S PRE-RUN PATTERN") print("=" * 60) hist = yf.download('OPEN', start=ONE_YEAR_AGO, end=TODAY, progress=False) if hist.empty: raise ValueError("No data for OPEN") if isinstance(hist.columns, pd.MultiIndex): if 'OPEN' in hist.columns.levels[1]: hist = hist.xs('OPEN', axis=1, level=1, drop_level=True) else: hist.columns = hist.columns.get_level_values(0) prices = hist['Close'] returns = prices.pct_change().dropna() rsi = compute_rsi(prices) atr = compute_atr(hist) fwd_30 = prices.pct_change(30).shift(-30) if fwd_30.dropna().empty: best_date = prices.index[-1] else: best_date = fwd_30.idxmax() print(f"\nOPEN's strongest 30-day run started around: {best_date.date()}") pre_run_start = best_date - timedelta(days=45) pre_run_end = best_date - timedelta(days=5) pre_hist = hist.loc[pre_run_start:pre_run_end] if len(pre_hist) < 10: print("Warning: limited pre-run data → using last 55 days") pre_hist = hist.iloc[-60:-5] pre_prices = pre_hist['Close'] pre_returns = pre_prices.pct_change().dropna() pre_rsi = compute_rsi(pre_prices) pre_atr = compute_atr(pre_hist) pre_vol_surge = volume_surge_score(pre_hist) pre_rsi_score = rsi_buildup_score(pre_rsi) pre_comp = compression_score(pre_hist, pre_atr) ann_vol = pre_returns.std() * np.sqrt(252) if len(pre_returns) > 1 else 0.0 print(f"\nOPEN Pre-Run Fingerprint:") print(f" Ann Vol : {ann_vol*100:.1f}%") print(f" Vol Surge : {pre_vol_surge:.2f}x") print(f" RSI Score : {pre_rsi_score:.2f}") print(f" Compression : {pre_comp:.2f}") if not pre_rsi.dropna().empty: print(f" RSI entry : {pre_rsi.dropna().iloc[-1]:.1f}") return { 'returns': returns, 'ann_vol': ann_vol, 'vol_surge': pre_vol_surge, 'rsi_score': pre_rsi_score, 'comp_score': pre_comp, } # ───────────────────────────────────────────── # SCAN SINGLE TICKER # ───────────────────────────────────────────── def scan_ticker(ticker, open_data): try: hist = yf.download(ticker, start=ONE_YEAR_AGO, end=TODAY, progress=False) if hist.empty or len(hist) < 60: return None if isinstance(hist.columns, pd.MultiIndex): if ticker in hist.columns.levels[1]: hist = hist.xs(ticker, axis=1, level=1, drop_level=True) else: hist.columns = hist.columns.get_level_values(0) prices = hist['Close'] returns = prices.pct_change().dropna() if len(returns) < 30: return None price = float(prices.iloc[-1]) ann_vol = returns.std() * np.sqrt(252) if len(returns) > 10 else 0.0 if not (PRICE_MIN <= price <= PRICE_MAX): return None if ann_vol < VOL_MIN_PCT / 100: return None open_ret = open_data['returns'] aligned_open = open_ret.reindex(returns.index).ffill().dropna() ret_aligned = returns.reindex(aligned_open.index).dropna() if len(ret_aligned) < 20: return None corr = float(ret_aligned.corr(aligned_open)) if np.isnan(corr) or corr < CORR_MIN: return None rsi = compute_rsi(prices) atr = compute_atr(hist) v_score = volume_surge_score(hist) r_score = rsi_buildup_score(rsi) c_score = compression_score(hist, atr) open_v = open_data['vol_surge'] open_r = open_data['rsi_score'] open_c = open_data['comp_score'] vol_match = max(0, 1 - abs(v_score - open_v) / max(open_v, 0.01)) rsi_match = max(0, 1 - abs(r_score - open_r)) cmp_match = max(0, 1 - abs(c_score - open_c)) composite = (vol_match * 0.4) + (rsi_match * 0.35) + (cmp_match * 0.25) beta = corr * (ann_vol / open_data['ann_vol']) if open_data['ann_vol'] > 0 else np.nan info = yf.Ticker(ticker).info or {} sector = info.get('sector') or info.get('industry') or 'N/A' mktcap = info.get('marketCap', None) mktcap_str = f"{mktcap/1e9:.2f}B" if mktcap and mktcap > 0 else 'N/A' cur_rsi = float(rsi.dropna().iloc[-1]) if not rsi.dropna().empty else np.nan return { 'Ticker': ticker, 'Price': round(price, 2), 'Vol%': round(ann_vol * 100, 1), 'Corr_OPEN': round(corr, 2), 'Beta_OPEN': round(beta, 2) if not np.isnan(beta) else 'N/A', 'Vol_Surge': round(v_score, 2), 'RSI': round(cur_rsi, 1) if not np.isnan(cur_rsi) else 'N/A', 'RSI_Score': round(r_score, 2), 'Compression': round(c_score, 2), 'Match_Score': round(composite * 100, 1), 'Market_Cap': mktcap_str, 'Sector': sector, } except: return None # ───────────────────────────────────────────── # MAIN # ───────────────────────────────────────────── def main(): try: open_data = get_open_fingerprint() except Exception as e: print(f"Failed to fingerprint OPEN: {e}") return tickers = get_russell_2000_tickers() if not tickers: print("No tickers available — exiting.") return filtered_tickers = pre_filter_tickers(tickers) print(f"\nScanning up to {len(filtered_tickers)} dynamic small-cap tickers...\n") results = [] for i, ticker in enumerate(sorted(set(filtered_tickers)), 1): if ticker == 'OPEN': continue row = scan_ticker(ticker, open_data) if row: results.append(row) print(f" ✓ {ticker:6s} ({i}/{len(filtered_tickers)}) Match: {row['Match_Score']:5.1f} RSI: {row['RSI']} VolSurge: {row['Vol_Surge']}x Compression: {row['Compression']:.2f}") if i % 20 == 0: print(f"Progress: {i}/{len(filtered_tickers)} processed...") if not results: print("\nNo qualifying matches found.") return df = pd.DataFrame(results).sort_values('Match_Score', ascending=False) print("\n" + "=" * 80) print("TOP MATCHES — Ranked by OPEN Pattern Similarity") print("=" * 80) print(df.head(20).to_string(index=False)) print("\n" + "=" * 80) print("🔥 HIGH CONVICTION (Match >50, Vol Surge >1.2, RSI 45-72)") print("=" * 80) high_conv = df[ (df['Match_Score'] > 50) & (df['Vol_Surge'] > 1.2) & (pd.to_numeric(df['RSI'], errors='coerce').between(45, 72)) ] if high_conv.empty: print("None right now — showing top 10 anyway:") print(df.head(10).to_string(index=False)) else: print(high_conv.to_string(index=False)) output_file = f"open_pattern_scan_dynamic_{TODAY}.csv" df.to_csv(output_file, index=False) print(f"\nSaved: {output_file}") if __name__ == "__main__": main()
